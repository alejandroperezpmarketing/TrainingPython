import math

## Elipsoide de Hayford
#Semieje mayor
a = 6378388
#Semieje menor
b = 6356911.94613


def calculos_previos():

    #Excentricidad
    e = math.sqrt(a*a - b*b)/a  
    ep = math.sqrt(a*a - b*b)/b
    ep2 = ep*ep
    
    #Radio polar de curvatura
    c = a*a/b
    
    return e, ep, ep2, c


#Eliminacin del retranqueo. 
#El retranqueo del eje de las X se realiza en todos los casos. 
#Para las Y, el retranqueo solamente se realiza en el caso de que estemos operando con coordenadas UTM del hemisferio sur

def eliminar_retranqueo(X, Y, hemisferio):
    X = X - 500000
    if hemisferio == 'S':
        Y = Y - 10000000
    return X, Y
